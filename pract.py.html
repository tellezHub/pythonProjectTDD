from tkinter import *
from math import *

def calcular():
    dato_altura = valorAltura.get()
    dato_radio = valorRadio.get()
    Res_calcular = round(pi * (dato_radio**2) * dato_altura,3)
    print("Respuesta: ", Res_calcular, "cm2")
    valorRespuesta.set(str(Res_calcular))

"""Creo la ventana"""

ven_calcular = Tk()
ven_calcular.title("Calcular el Volumen de Cilindro")
ven_calcular.geometry("650x250")
"""_________________________________________________"""

titulo = Label(ven_calcular, text="CALCULAR El VOLUMEN DE UN CILINDRO", width=50, height=2, bg="red", fg="black")
titulo.place(x=50,y=15)

espa = Label(ven_calcular, text=" ",height=5 )
espa.grid(column=1, row=2)

altura = Label(ven_calcular, text="Ingrese la altura: ", fg="blue")
altura.grid(column=1, row=3)

radio = Label(ven_calcular, text="Ingrese el radio: ", fg="blue")
radio.grid(column=1, row=4)

resp = Label(ven_calcular, text="RESPUESTA:", fg="black")
resp.place(x=20, y=140)

cm3 = Label(ven_calcular, text="CM3", fg="black")
cm3.place(x=210, y=140)

valorAltura = DoubleVar()
Caja_altura = Entry(ven_calcular, width=20, textvariable=valorAltura)
Caja_altura.grid(column=2, row=3)

valorRadio = DoubleVar()
Caja_radio = Entry(ven_calcular, width=20, textvariable=valorRadio)
Caja_radio.grid(column=2, row=4)

valorRespuesta = DoubleVar()
Caja_res = Entry(ven_calcular, width=17, textvariable=valorRespuesta)
Caja_res.place(x=100, y=140)

boton_cal = Button(ven_calcular, text="CALCULAR", fg="white", bg="black", width=15, height=2, command=calcular)
boton_cal.place(x=260, y=80)

boton_salir = Button(ven_calcular, text="SALIR", fg="white", bg="black", width=15, height=2, command=quit)
boton_salir.place(x=260, y=150)

"""______________________IMAGEN___________________________"""
imagen_cilindro = PhotoImage(file="cilindro.gif")
etiqueta_foto = Label(ven_calcular, image=imagen_cilindro, width=250, height=300)
etiqueta_foto.image = imagen_cilindro
etiqueta_foto.place(x=400, y=5)

ven_calcular.mainloop()
